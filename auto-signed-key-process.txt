# Générer la clé privée
openssl genpkey -algorithm RSA -out private_key.pem

# Générer la demande de certificat avec une période de validité très longue (par exemple, 10000 jours)
openssl req -new -key private_key.pem -out certificate_request.csr -days 10000

# Auto-signer le certificat avec une période de validité très longue
openssl x509 -req -in certificate_request.csr -signkey private_key.pem -out certificate.pem -days 10000

# Convertir en JKS pour Android
keytool -importcert -v -trustcacerts -file certificate.pem -alias myalias -keystore keystore.jks -storepass your_keystore_password
